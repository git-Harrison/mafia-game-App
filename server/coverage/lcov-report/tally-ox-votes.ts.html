// VOTE_OX 페이즈 종료 시점의 영역 점유를 집계해 처형 여부를 판정한다.
// 룰: 후보 본인은 자동 X 1표 pre-count (서버 권위) — 후보의 O 점유는 무시.
//     영역 밖 (입력 map 미포함) = 기권.
//     정확히 절반 = 부결, 과반 = 처형 (oCount * 2 > aliveCount).
// (자체 작성 정책 — mafia-app-context.md §8-1 룰 1 / VOTE_OX 분할 룰)
 
import type { OxTallyInput, OxTallyResult } from './types';
 
export function tallyOxVotes(input: OxTallyInput): OxTallyResult {
  const { ox, defendantUserId, aliveUserIds } = input;
  const aliveSet = new Set(aliveUserIds);
  const aliveCount = aliveUserIds.length;
 
  let oCount = 0;
  let xCount = 0;
 
  for (const [userId, vote] of ox) {
    if (!aliveSet.has(userId)) continue; // 생존자 외 입력은 방어적 무시
    if (userId === defendantUserId) continue; // 후보는 강제 X 로 별도 처리
    // OxVote 타입은 'O' | 'X' 두 값뿐이므로 단일 분기로 정리.
    if (vote === 'O') oCount++;
    else xCount++;
  }
 
  // 후보 자동 X 1표 — 입력 map 의 후보 점유와 무관하게 강제 1표.
  if (aliveSet.has(defendantUserId)) {
    xCount++;
  }
 
  const executed = oCount * 2 > aliveCount;
 
  return { executed, oCount, xCount };
}
